cmake_minimum_required(VERSION 3.18)

project(raytracer LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release)
endif()

# ---- stb implementation files (compiled exactly once) ----
# These dedicated TUs define STB_IMAGE_IMPLEMENTATION / STB_IMAGE_WRITE_IMPLEMENTATION
# so that no other source file needs to. Eliminates ODR violations.
set(STB_SOURCES
    external/stb_image_impl.cpp
    external/stb_image_write_impl.cpp
)

# ---- shared source files ----
set(SHARED_SOURCES
    core/vec3.cpp
    core/ray.cpp
    core/camera.cpp
    objects/sphere.cpp
    objects/triangle.cpp
    objects/plane.cpp
    objects/bvh_node.cpp
    objects/hittable_list.cpp
    objects/obj_loader.cpp
    materials/lambertian.cpp
    materials/metal.cpp
    materials/dielectric.cpp
    materials/glossy.cpp
    textures/image_texture.cpp
    lights/point_light.cpp
    lights/directional_light.cpp
    lights/area_light.cpp
    scenes/scene_registry.cpp
    scenes/scene_random_spheres.cpp
    scenes/scene_normal_map_test.cpp
    scenes/scene_sponza.cpp
    scenes/scene_san_miguel.cpp
)

# ---- CPU target ----
add_executable(render
    src/main.cpp
    ${SHARED_SOURCES}
    ${STB_SOURCES}
)
target_include_directories(render PRIVATE .)
target_compile_options(render PRIVATE
    $<$<CXX_COMPILER_ID:MSVC>:/O2 /W3>
    $<$<NOT:$<CXX_COMPILER_ID:MSVC>>:-O3 -march=native>
)
find_package(Threads REQUIRED)
target_link_libraries(render PRIVATE Threads::Threads)

# ---- CUDA target ----
include(CheckLanguage)
check_language(CUDA)

if(CMAKE_CUDA_COMPILER)
    enable_language(CUDA)
    set(CMAKE_CUDA_STANDARD 17)

    if(NOT DEFINED CMAKE_CUDA_ARCHITECTURES)
        set(CMAKE_CUDA_ARCHITECTURES 86)
    endif()

    add_executable(render_cuda
        src/main_cuda.cu
        renderer/cuda/cuda_renderer.cu
        ${SHARED_SOURCES}
        ${STB_SOURCES}
    )

    target_include_directories(render_cuda PRIVATE .)

    set_target_properties(render_cuda PROPERTIES
        CUDA_SEPARABLE_COMPILATION ON
    )

    target_compile_options(render_cuda PRIVATE
        $<$<COMPILE_LANGUAGE:CUDA>:
            --use_fast_math
            --expt-relaxed-constexpr
        >
        $<$<AND:$<COMPILE_LANGUAGE:CXX>,$<CXX_COMPILER_ID:MSVC>>:
            /O2 /W3
        >
    )

    target_link_libraries(render_cuda PRIVATE
        Threads::Threads
        curand
    )

    message(STATUS "CUDA ${CMAKE_CUDA_VERSION} found — render_cuda target enabled")
    message(STATUS "CUDA architectures: ${CMAKE_CUDA_ARCHITECTURES}")
else()
    message(STATUS "CUDA not found — only CPU render available")
endif()
